//Filter names starting with A



const checkName = (users) => {
    var item = false ;
    if (users.name.includes("A")){
        item= true;
    }
    return item;    
};


const users = [
 { name: "Alice" },
 { name: "Bob" },
 { name: "Adam" }
];


const NamesWithA = users.filter(checkName);

console.log("Result :", NamesWithA);
