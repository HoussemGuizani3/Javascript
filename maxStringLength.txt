function maxStringLength(t) {
    var max = t[0].length; 
    
    for (var i = 1; i < t.length; i++) {
        if (t[i].length > max) {
            max = t[i].length;
        }
    }
    return max;
}

var n = prompt("The size of the table? ");
const t = [];

for (var i = 0; i < n; i++) {
    var ch = prompt("Enter item number n " + (i + 1) + " : ");
    t.push(ch); 
}
