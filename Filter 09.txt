// Filter completed orders


const checkOrders = (orders) => {
    var item = false ;
    if (orders.status === "completed"){
        item= true;
    }
    return item;    
};


const orders = [
 { id: 1, status: "pending" },
 { id: 2, status: "completed" },
 { id: 3, status: "completed" }
];



const CompletedOrders = orders.filter(checkOrders);

console.log("Result :", CompletedOrders);
