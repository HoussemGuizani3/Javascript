// Filter products in stock

const filterProductWithStock = (product) => {
    var item= false ;
    if (product.stock > 0) {
        item= true;
    }
    return item;
};

const products = [
 { name: "Laptop", stock: 0 },
 { name: "Phone", stock: 5 },
 { name: "Tablet", stock: 2 }
];


const stockBigerThanZero = products.filter(filterProductWithStock);

console.log("En Stock :", stockBigerThanZero);
